CONSULTA 5

SELECT DISTINCT ?film ?filmLabel
WHERE
{
   ?film wdt:P86 wd:Q76364  
   
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE]". }
}
ORDER BY ASC (?filmLabel)

CONSULTA 6

WHERE
{
   ?film wdt:P86 wd:Q76364  
   
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE]". }
}
HAVING (COUNT(?film))
